## Creating a Framework Instance

```ts
import { Framework } from "@aspen-os/framework/server";
import { OrganizationModule } from "@aspen-os/organization";
import { ComplianceModule } from "@aspen-os/compliance";

const organization = OrganizationModule.create({ country: "INDIA" });
const compliance = ComplianceModule.create({ country: "INDIA" });

const framework = Framework.create(
  {
    auth: authConfig,
    db: dbConfig,
    kvStore: kvConfig,
    logs: logConfig,
    pubsub: pubsubConfig,
    rpc: rpcConfig,
    storage: storageConfig,
  },
  { organization, compliance },
);

await framework.prepare();
```

## Running in Request Context

The `run()` method provides request-scoped context via `AsyncLocalStorage`:

```ts
import { getContext } from "@aspen-os/framework/server";

await framework.run(async () => {
  const { db, pubsub } = getContext();
  // db: NodePgDatabase (drizzle)
  // pubsub: PubSubUnit

  const result = await db.select().from(organizations);
  await pubsub.publish("organization:created", { organization });
});
```

## Using the Auth Unit

```ts
// Create a user
const user = await framework.auth.user.create({
  email: "user@example.com",
  password: "secure-password",
  name: "John Doe",
});

// Create a session
const { user, session } = await framework.auth.session.create(
  "user@example.com",
  "secure-password",
);

// Validate a session
const result = await framework.auth.session.validate(token);

// Assign a role
await framework.auth.user.role.assign(userId, "admin");
```

## Using the Storage Unit

```ts
// Upload a file
const file = await framework.storage.upload({
  key: "documents/report.pdf",
  body: buffer,
  contentType: "application/pdf",
  metadata: { uploadedBy: userId },
});

// Get a signed URL
const url = await framework.storage.getSignedGetUrl("documents/report.pdf", {
  expiresIn: 3600,
});

// Archive a file
await framework.storage.archive("documents/report.pdf");
```

## Using the KV Store

```ts
// Set with TTL
await framework.kvStore.set("cache:key", { data: "value" }, 1800);

// Get (returns null if expired)
const value = await framework.kvStore.get("cache:key");

// Get or set (cache-aside pattern)
const result = await framework.kvStore.getOrSet(
  "expensive:query",
  async () => await runExpensiveQuery(),
  3600,
);

// Increment/decrement counters
await framework.kvStore.increment("visits");
await framework.kvStore.decrement("remaining");
```

## Using PubSub

```ts
// Publish a message
await framework.pubsub.publish("organization:created", { organization });

// Subscribe to a topic
await framework.pubsub.subscribe("organization:created", async (message) => {
  console.log("Received:", message.data);
});

// Schedule a recurring job
await framework.pubsub.schedule(
  "compliance:weekly-summary",
  "0 9 * * 1",  // every Monday at 9am
  {},
  { tz: "Asia/Kolkata" },
);
```
